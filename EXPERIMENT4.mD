7Ô∏è‚É£ Display ename, sal, hra, da, pf, totalsal

HRA = 15% of salary

DA = 10% of salary

PF = 5% of salary

Total Salary = (sal + hra + da) - pf

SELECT 
    ename,
    sal,
    sal*0.15 AS hra,
    sal*0.10 AS da,
    sal*0.05 AS pf,
    (sal + sal*0.15 + sal*0.10 - sal*0.05) AS totalsal
FROM employee
ORDER BY totalsal;
